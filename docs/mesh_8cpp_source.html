152 struct triangulateio Triangle::tethex_to_io(tethex::Mesh &mesh) const
153 {
154  struct triangulateio io;
155  set_tria_to_default(&io);
156 
157  //Points
158  if(!mesh.vertices.empty())
159  {
160  io.pointlist = new double[2 * mesh.vertices.size()];
161  io.pointmarkerlist = new int[mesh.vertices.size()];
162  io.numberofpoints = mesh.vertices.size();
163 
164  int i = 0;
165  for(auto &p: mesh.vertices)
166  {
167  io.pointlist[2 * i ] = p.get_coord(0);
168  io.pointlist[2 * i + 1] = p.get_coord(1);
169  io.pointmarkerlist[i] = p.regionTag;
170  ++i;
171  }
172  }
173 
174  //Segments
175  if(!mesh.lines.empty())
176  {
177  io.segmentlist = new int[2 * mesh.lines.size()];
178  io.segmentmarkerlist = new int[mesh.lines.size()];
179  io.numberofsegments = mesh.lines.size();
180  int i = 0;
181  for(auto l: mesh.lines)
182  {
183  io.segmentlist[2 * i ] = l->get_vertex(0) + 1;//HOTFIX FIXME
184  io.segmentlist[2 * i + 1] = l->get_vertex(1) + 1;//HOTFIX FIXME
185  io.segmentmarkerlist[i] = l->get_material_id();
186  ++i;
187  }
188  }
189 
190  if(!mesh.triangles.empty())
191  {
192  io.trianglelist = new int[3 * mesh.lines.size()];
193  io.numberoftriangleattributes = 1;
194  io.triangleattributelist = new double[mesh.lines.size()];
195  io.numberofsegments = mesh.lines.size();
196  int i = 0;
197  for(auto t: mesh.triangles)
198  {
199  io.trianglelist[3 * i ] = t->get_vertex(0);
200  io.trianglelist[3 * i + 1] = t->get_vertex(1);
201  io.trianglelist[3 * i + 2] = t->get_vertex(2);
202  io.triangleattributelist[i] = t->get_material_id();
203  ++i;
204  }
205  }
206 
207  return io;
208 }

210  void Triangle::io_to_tethex(
211  struct triangulateio &io, tethex::Mesh &initMesh)
212 {
213  vector<tethex::Point> pointsVal;
214  vector<tethex::MeshElement*> segmentsVal;
215  vector<tethex::MeshElement*> trianglesVal;
216 
217  pointsVal.reserve(io.numberofpoints);
218  for(int i = 0; i < io.numberofpoints; ++i)
219  {
220  auto x = io.pointlist[2 * i],
221  y = io.pointlist[2 * i + 1];
222  auto regionTag = io.pointmarkerlist[i];
223 
224  pointsVal.push_back(tethex::Point(x, y, 0/*z-component*/, regionTag, 1./*default mesh size used in gmsh*/));
225  }
226 
227  segmentsVal.reserve(io.numberofsegments);
228  for(int i = 0; i < io.numberofsegments; ++i)
229  {
230  auto v1 = io.segmentlist[2*i] - 1,//NOTE vertices index should start from zero
231  v2 = io.segmentlist[2*i + 1] - 1,
232  regionTag = io.segmentmarkerlist[i];
233  if(regionTag == 0)//NOTE boundary_id of inner cells in Deal.II should be -1
234  regionTag = -1;
235  segmentsVal.push_back(new tethex::Line(v1, v2, regionTag));
236  }
237 
238  trianglesVal.reserve(io.numberoftriangles);
239  for(int i = 0; i < io.numberoftriangles; ++i)
240  {
241  auto v1 = io.trianglelist[3*i] - 1,//NOTE vertice index should start from zero
242  v2 = io.trianglelist[3*i + 1] - 1,
243  v3 = io.trianglelist[3*i + 2] - 1;
244 
245  //taking only the first one attribute
246  double regionTag = 0;
247  if(io.numberoftriangleattributes != 0)
248  regionTag = io.triangleattributelist[io.numberoftriangleattributes * i];
249 
250  trianglesVal.push_back(new tethex::Triangle(v1, v2, v3, regionTag));
251  }
252 
253  initMesh.set_vertexes(pointsVal);
254  initMesh.set_lines(segmentsVal);
255  initMesh.set_triangles(trianglesVal);
256 }

299 string Triangle::update_options()
300 {
301  options = "";
302  #define MAX_ANGLE 36
303  if (ConstrainAngle) options += "q";

307  throw invalid_argument("Triangle quality angle should be equal or less then 36");
308  //if (StartNumberingFromZero)

311  if (AreaConstrain) options += "a";
312  if (AreaConstrain && MaxTriaArea > 0)
313  options += to_string(MaxTriaArea);
314  if (CustomConstraint) options += "u";
315  if (DelaunayTriangles) options += "D";
316  if (EncloseConvexHull) options += "c";
317  if (CheckFinalMesh) options += "C";
318  if (AssignRegionalAttributes) options += "A";
319  if (VoronoiDiagram) options += "v";
320  if (Quite) options += "Q";
321  else if (Verbose) options += "V";

323  options += "F";
324  else if (Algorithm == ITERATOR)

328  options += "B";
329  if (SuppressPolyFile) options += "P";
330  if (SuppressNodeFile) options += "N";
331  if (SuppressEleFile) options += "E";
332  if (OutputEdges) options += "e";
333  if (ComputeNeighbours) options += "n";
334  if (SuppressMehsFileNumbering)
335  options += "I";
336  if (SuppressExactArithmetics)
337  options += "X";
338  if (SuppressHoles) options += "O";
339  //if (SecondOrderMesh)
340  // options += "o2";
341  if (SteinerPointsOnBoundary) options += "Y";
342  if (SteinerPointsOnSegments) options += "YY";
343  if (MaxNumOfSteinerPoints > 0)
344  options += "S" + to_string(MaxNumOfSteinerPoints);
