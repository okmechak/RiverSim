230  {

232  QGauss<dim - 1> face_quadrature_formula(fe.degree + 1);

240  fe,

244  const unsigned dofs_per_cell = fe.dofs_per_cell;
245  const unsigned n_q_points = quadrature_formula.size();
246  const unsigned n_face_q_points = face_quadrature_formula.size();
247 
248  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
249  Vector<double> cell_rhs(dofs_per_cell);
250 
251  vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
252 
253  for(const auto& cell: dof_handler.active_cell_iterators())
254  {
255  fe_values.reinit(cell);
256 
257  cell_matrix = 0;
258  cell_rhs = 0;
259 
260  for(unsigned q_index = 0; q_index < n_q_points; ++q_index)
261  for(unsigned i = 0; i < dofs_per_cell; ++i)
262  {
263  for(unsigned j = 0; j < dofs_per_cell; ++j)
264  cell_matrix(i, j) +=
265  fe_values.shape_grad(i, q_index) *
266  fe_values.shape_grad(j, q_index) *
267  fe_values.JxW(q_index);
268 
269  cell_rhs(i) += fe_values.shape_value(i, q_index) *
270  1 *
271  fe_values.JxW(q_index);
272  }
273 
274  for(unsigned face_number = 0; face_number < GeometryInfo<dim>::faces_per_cell; ++face_number)
275  if(cell->face(face_number)->at_boundary() && cell->face(face_number)->boundary_id() == 2)
276  {
277  fe_face_values.reinit(cell, face_number);
278 
279  for(unsigned q_point = 0; q_point < n_face_q_points; ++q_point)
280  {

282  for(unsigned i = 0; i < dofs_per_cell; ++i)
283  cell_rhs(i) +=
284  neuman_value *
285  fe_face_values.shape_value(i, q_point)*
286  fe_face_values.JxW(q_point);
287 
288  }
289  }
290 
291  cell->get_dof_indices(local_dof_indices);
292  constraints.distribute_local_to_global(
293  cell_matrix,
294  cell_rhs,

319  vector<double> series_param_integral(const River::Point point, const double angle)
320  {
321  QGauss<dim> quadrature_formula(fe.degree);
322 
323  FEValues<dim> fe_values(
324  fe, quadrature_formula,
325  update_values |
326  update_JxW_values);
327 
328  vector<double>
329  values(quadrature_formula.size()),
330  integral(3, 0),
331  normalization_integral(3, 0),
332  series_params(3, 0); //Series params
333 
334  cout << "point.... " << point << endl;
335 
336  for(auto dof_cell: dof_handler.active_cell_iterators())
337  {
338  //central point of mesh element
339  auto center = dof_cell->center();
340 
341  //distance between tip point and current mesh element
342  auto dx = center[0] - point[0],
343  dy = center[1] - point[1],
344  dist = sqrt(dx*dx + dy*dy);
345 
346  //Integrates over points only in this circle
347  if(dist <= model->integr.integration_radius)
348  {
349  fe_values.reinit (dof_cell);
350  fe_values.get_function_values(solution, values);
351  auto weight_func_value = model->integr.WeightFunction(dist);
352 
353  //cycle over all series parameters order
354  for(unsigned param_index = 0; param_index < series_params.size(); ++param_index)
355  {
356  //preevaluate basevector value
357  auto base_vector_value = model->integr.BaseVectorFinal(param_index + 1, angle, dx, dy);
358 
359  //sum over all quadrature points overIntegrationRadius single mesh element
360  for (unsigned q_point = 0; q_point < quadrature_formula.size(); ++q_point)
361  {
362  //integration of weighted integral..
363  integral[param_index]
364  += values[q_point]
365  * weight_func_value
366  * base_vector_value
367  * fe_values.JxW(q_point);
368 
369  //.. and its normalization integral
370  normalization_integral[param_index]
371  += weight_func_value
372  * pow(base_vector_value, 2)
373  * fe_values.JxW(q_point);
374  }
375  }
376  }
377  }
378  for(unsigned i = 0; i < integral.size(); ++i)
379  series_params[i] = integral[i]/normalization_integral[i];
380 
381  return series_params;
382  }

407  {

420  max_a = max_a_backward;
421  else
422  for(auto&[id, series_params]: id_series_params)
423  if (max_a < series_params.at(0))
424  max_a = series_params.at(0);
425 
426  for(auto&[id, series_params]: id_series_params)
427  if(model->q_growth(series_params))

433  auto br_left = Branch(tip_point, tip_angle + model->bifurcation_angle);

435  auto br_right = Branch(tip_point, tip_angle - model->bifurcation_angle);
436  br_right.AddPoint(Polar{model->ds, 0});
437  model->tree.AddSubBranches(id, br_left, br_right);
438  }
