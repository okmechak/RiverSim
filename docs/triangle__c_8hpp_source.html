114  bool refinementFunction(const River::Point & p1, const River::Point & p2, const River::Point & p3, const double &area)
115  {
116  // evaluatio of edge lenghts
117  // lengths are of the first priority
118  auto
119  dl1 = (p1-p2).norm(),
120  dl2 = (p2-p3).norm(),
121  dl3 = (p3-p1).norm();
122 
123  auto max_dl = dl1;
124  if (max_dl < dl2)
125  max_dl = dl2;
126  if (max_dl < dl3)
127  max_dl = dl3;
128 
129  if (max_dl > max_edge)
130  return true;
131 
132  auto min_dl = dl1;
133  if (min_dl > dl2)
134  min_dl = dl2;
135  if (min_dl > dl3)
136  min_dl = dl3;
137 
138  auto cur_ratio = max_dl / min_dl;
139  if (cur_ratio > ratio && min_dl > min_edge)
140  return true;
141 
142  if (area > max_area && min_dl > min_edge)
143  return true;
144 
145  // test of adaptive mesh function
146  REAL
147  vert_val_1 = meshAreaConstraint(p1.x, p1.y),
148  vert_val_2 = meshAreaConstraint(p2.x, p2.y),
149  vert_val_3 = meshAreaConstraint(p3.x, p3.y);
150  if ((area > vert_val_1 || area > vert_val_2 || area > vert_val_3)
151  && min_dl > min_edge && area > min_area)
152  return true;
153 
154  //constrain triangle area in better way
155  for(const auto& tip_point: tip_points)
156  if (PointInTriangle(tip_point, p1, p2, p3) && meshAreaConstraint(tip_point.x, tip_point.y) < area
157  && min_dl > min_edge && area > min_area)
158  return true;
159 
160  return false;
161  };

165  inline double meshAreaConstraint(double x, double y) const
166  {
167  double result_area = max_area;
168  for (auto &tip : tip_points)
169  {
170  auto r = (River::Point{x, y} - tip).norm();
171  auto exp_val = exp(-pow(r / refinment_radius, exponant) / 2. / sigma / sigma);
172  auto cur_area = min_area + (max_area - min_area) * (1. - exp_val) / (1. + exp_val);
173  if (result_area > cur_area)
174  result_area = cur_area;
175  }
176 
177  return result_area;
178  }
